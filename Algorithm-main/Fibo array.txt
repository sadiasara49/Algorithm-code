#include<iostream>
using namespace std;
int fibo(int n)
{
   int A[10];
   A[0]=0;
   A[1]=1;
   for(int i=2;i<=n;i++)
   {

      A[i]=A[i-1]+A[i-2];

   }

  return A[n] ;

}
int main()
{
  int n;
  cin>>n;
  cout<<"Series:";
  for(int i=0;i<n;i++)
  {

    cout<<fibo(i)<<" ";

  }

   cout<<endl;
   cout<<fibo(n) ;

}
